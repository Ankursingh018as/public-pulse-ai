#### Inference Pipeline
```python
# main.py
from fastapi import FastAPI
from src.classifier import CivicClassifier
from src.predictor import TimeSeriesPredictor
from src.fusion_engine import FusionEngine

app = FastAPI(title="Public Pulse AI Engine")

# Load models at startup
classifier = CivicClassifier.load("models/classifier.pkl")
predictor = TimeSeriesPredictor.load_all("models/")
fusion = FusionEngine(classifier, predictor)

@app.post("/classify")
async def classify_incident(request: ClassifyRequest):
    result = classifier.predict(request.text)
    severity = fusion.calculate_severity(result, request.location, request.time)
    return {
        "category": result.category,
        "severity": severity,
        "confidence": result.confidence,
        "sub_category": result.sub_category
    }

@app.post("/predict")
async def generate_predictions(request: PredictRequest):
    predictions = predictor.forecast(
        area_id=request.area_id,
        horizon=request.horizon,
        types=request.prediction_types
    )
    return {"predictions": predictions}

@app.post("/anomaly")
async def detect_anomalies(request: AnomalyRequest):
    anomalies = fusion.detect_anomalies(request.readings)
    return {"anomalies": anomalies, "alert": len(anomalies) > 0}
```

-- Performance indexes
CREATE INDEX idx_incidents_status ON incidents(status) WHERE status = 'active';

CREATE INDEX idx_incidents_created ON incidents(created_at DESC);

-- TimescaleDB continuous aggregates
CREATE MATERIALIZED VIEW hourly_incidents
WITH (timescaledb.continuous) AS
SELECT 
    time_bucket('1 hour', created_at) AS hour,
    area_id,
    event_type,
    COUNT(*) as count,
    AVG(severity) as avg_severity
FROM incidents
GROUP BY hour, area_id, event_type;

#### Cache Patterns
```typescript
// Cache-aside pattern for incident list
async function getActiveIncidents(): Promise<Incident[]> {
    const cacheKey = 'incidents:active';
    
    // 1. Check cache
    const cached = await redis.get(cacheKey);
    if (cached) {
        return JSON.parse(cached);
    }
    
    // 2. Cache miss - query database
    const incidents = await db.query(`
        SELECT * FROM incidents 
        WHERE status = 'active' 
        ORDER BY severity DESC, created_at DESC
        LIMIT 100
    `);
    
    // 3. Store in cache
    await redis.setex(cacheKey, 30, JSON.stringify(incidents));
    
    // 4. Return data
    return incidents;
}

// Cache invalidation on write
async function createIncident(incident: Incident): Promise<void> {
    await db.insert('incidents', incident);
    
    // Invalidate related caches
    await redis.del('incidents:active');
    await redis.del(`incidents:area:${incident.area_id}`);
    await redis.del('analytics:summary');
}
```
